tcp.c
struct tcp_pcb *
tcp_alloc(u8_t prio)

lwipopts.h
#define  TCP_KEEPIDLE_DEFAULT     7200000UL//2h后第一次发送保活包
#define  TCP_KEEPINTVL_DEFAULT    75000UL//每隔75s发送一个保活包
#define  TCP_KEEPCNT_DEFAULT      9U//连续9次没有收到对方回应，认为连接已断开
